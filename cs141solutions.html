#factorization
def factorize(x) :
  factors=[]
  t=2
  while t<=x :
    if x%t==0 :
      factors.append(t)
    t+=1
  return factors

#prime factors
def prime(l) :
  prime_factors=[]
  i=0
  while (i <len(l)) :
    x=l[i]
    t = 2
    while (t <x):
      if (x%t == 0):
        break
      t = t + 1
    if (t == x):
      prime_factors.append(x)
    i+=1
  return prime_factors

print("factors of",n,"are",prime(factorize(n)))

# powers of prime factors
def powers(l) :
  powers=[]
  i=0
  while i<len(l)) :
    x=l[i]
    j=0
    while n%(x**(j+1))==0 :
      j+=1
    powers.append(j)
    j=0
    i+=1
  return powers

print("powers corrosponding to the prime factors", powers(prime(factorize(n))))

def Ds(D) :
  T=[]
  S=[]
  SD={}
  for k,v in D.items():
    T.append(v)
  S=sorted(T, reverse=True)
  for s in S :
    for k,v in D.items() :
      if v==s :
        SD[k]=v
  return(SD)
